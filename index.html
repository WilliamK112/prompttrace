现在，我们在 apiCalls.js 文件中添加日志中间件，以记录 API 调用的信息：

Now we add log middleware to the apiCalls.js file to record the information of the API call: 

const express = require('express');
const app = express();

// 添加日志中间件
app.use((req, res, next) => {
  const startTime = Date.now();
  console.log(`API Call: ${req.method} ${req.url} - Start Time: ${startTime}`);

  res.on('finish', () => {
    const endTime = Date.now();
    console.log(`API Call: ${req.method} ${req.url} - End Time: ${endTime} - Duration: ${endTime - startTime}ms`);
  });

  next();
});

// 示例 API 调用
app.get('/api/data', (req, res) => {
  res.send({ data: 'some data' });
});

app.listen(3000, () => {
  console.log('Server is running on port 3000');
});
